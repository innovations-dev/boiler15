## Header Component

The header component is a key part of our application's navigation system. It's responsive, supports dark mode, and includes authentication state.

### Basic Header

```tsx
import { Header } from "@/components/header";

export default function Page() {
  return (
    <Header>
      <Header.Logo />
      <Header.Nav />
      <Header.Auth />
    </Header>
  );
}
```

### With Custom Navigation

```tsx
import { Header } from "@/components/header";

const navigation = [
  { name: "Dashboard", href: "/dashboard" },
  { name: "Team", href: "/team" },
  { name: "Projects", href: "/projects" },
];
export default function CustomHeader() {
  return (
    <Header>
      <Header.Logo />
      <Header.Nav items={navigation} />
      <Header.Actions>
        <Header.ThemeToggle />
        <Header.Auth />
      </Header.Actions>
    </Header>
  );
}
```

### Mobile Responsive Menu

```tsx
import Link from "next/link";
import { Menu } from "lucide-react";

import { Header } from "@/components/header";
import { Button } from "@/components/ui/button";
import { Sheet } from "@/components/ui/sheet";

export function MobileHeader() {
  return (
    <Header>
      <Header.Logo />
      <Header.MobileNav>
        <Sheet>
          <Sheet.Trigger asChild>
            <Button variant="ghost" size="icon">
              <Menu className="h-5 w-5" />
            </Button>
          </Sheet.Trigger>
          <Sheet.Content side="left">
            <Sheet.Header>
              <Sheet.Title>Navigation</Sheet.Title>
            </Sheet.Header>
            <nav className="grid gap-4 py-4">
              {navigation.map((item) => (
                <Link
                  key={item.href}
                  href={item.href}
                  className="text-sm font-medium"
                >
                  {item.name}
                </Link>
              ))}
            </nav>
          </Sheet.Content>
        </Sheet>
      </Header.MobileNav>
    </Header>
  );
}
```

## Form Components

Our form components are built with accessibility and validation in mind.

### Basic Form

```tsx
import { Button } from "@/components/ui/button";
import { Form } from "@/components/ui/form";
import { Input } from "@/components/ui/input";

export function SignInForm() {
  return (
    <Form {...form}>
      <Form.Field
        control={form.control}
        name="email"
        render={({ field }) => (
          <Form.Item>
            <Form.Label>Email</Form.Label>
            <Form.Control>
              <Input placeholder="email@example.com" {...field} />
            </Form.Control>
            <Form.Message />
          </Form.Item>
        )}
      />
      <Button type="submit">Sign In</Button>
    </Form>
  );
}
```

### With Validation

```tsx
import { zodResolver } from "@hookform/resolvers/zod";
import { useForm } from "react-hook-form";
import as z from "zod";
const formSchema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
});
export function ValidatedForm() {
  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
  });

  return (
    <Form {...form}>
      <Form.Field
        control={form.control}
        name="email"
        render={({ field }) => (
          <Form.Item>
            <Form.Label>Email</Form.Label>
            <Form.Control>
              <Input {...field} />
            </Form.Control>
            <Form.Message />
          </Form.Item>
        )}
      />
    </Form>
  );
}
```

## Dialog Components

Modal dialogs for user interactions.

### Basic Dialog

```tsx
import { Button } from "@/components/ui/button";
import { Dialog } from "@/components/ui/dialog";

export function ConfirmDialog() {
  return (
    <Dialog>
      <Dialog.Trigger asChild>
        <Button variant="outline">Delete Account</Button>
      </Dialog.Trigger>
      <Dialog.Content>
        <Dialog.Header>
          <Dialog.Title>Are you sure?</Dialog.Title>
          <Dialog.Description>This action cannot be undone.</Dialog.Description>
        </Dialog.Header>
        <Dialog.Footer>
          <Button variant="ghost">Cancel</Button>
          <Button variant="destructive">Delete</Button>
        </Dialog.Footer>
      </Dialog.Content>
    </Dialog>
  );
}
```

## Best Practices

1. **Component Organization**

   - Keep components small and focused
   - Use composition over inheritance
   - Implement proper TypeScript interfaces

2. **Accessibility**

   - Include proper ARIA labels
   - Support keyboard navigation
   - Test with screen readers

3. **Responsive Design**
   - Use Tailwind's responsive modifiers
   - Test on multiple screen sizes
   - Implement proper mobile interactions
